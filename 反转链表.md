# [反转链表](https://leetcode.cn/problems/reverse-linked-list/)

```js
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var reverseList = function(head) {
    let dummyNode = new ListNode()
    let cur = head
    while(cur !== null){
        let preNode = dummyNode.next
        let temp = cur.next
        dummyNode.next = cur
        cur.next = preNode
        cur = temp
    }
    return dummyNode.next
};
```
