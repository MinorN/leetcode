# [回文链表](https://leetcode.cn/problems/palindrome-linked-list/description/)

```js
var isPalindrome = function (head) {
    // 找到中间节点
    // 快慢指针找到中间节点
    let slow = head, fast = head;
    while (fast && fast.next) {
        slow = slow.next; // slow 一次走一步
        fast = fast.next.next; // fast 一次走两步
    }
    // 反转后半部分链表
    let prev = null;
    let current = slow;
    while (current) {
        let next = current.next;
        current.next = prev;
        prev = current;
        current = next;
    }
    // 比较前半部分和反转后的后半部分
    let left = head;
    let right = prev;
    while (right) {
        if (left.val !== right.val) {
            return false;
        }
        left = left.next;
        right = right.next;
    }

    return true;

};
```
